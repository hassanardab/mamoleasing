  const [eventStartDate, setEventStartDate] = useState<Date | undefined>(
    new Date()
  );
  const [selectedEvent, setSelectedEvent] = useState<BookingEvent | null>(null);
  const [isEventDetailsOpen, setIsEventDetailsOpen] = useState(false);
  const [selectedUnconfirmedEvent, setSelectedUnconfirmedEvent] =
    useState<CombinedEvent | null>(null);
  const [isOnlineBookingDialogOpen, setIsOnlineBookingDialogOpen] =
    useState(false);
  const [isPendingCustomersDialogOpen, setIsPendingCustomersDialogOpen] =
    useState(false);
  const [isEventsListDialogOpen, setIsEventsListDialogOpen] = useState(false);
  const [eventsForDialog, setEventsForDialog] = useState<CombinedEvent[]>([]);
  const [eventDialogTitle, setEventDialogTitle] = useState("");
  const [eventDialogDescription, setEventDialogDescription] = useState("");
  const [isMigrating, setIsMigrating] = useState(false);
  const [isMigrationNeeded, setIsMigrationNeeded] = useState(false);


  const [editingEvent, setEditingEvent] = useState<BookingEvent | null>(null);
  const [isEditDialogOpen, setIsEditDialogOpen] = useState(false);


  // New state for enrichment
  const [journalEntries, setJournalEntries] = useState<JournalEntry[]>([]);
  const [cashAccountId, setCashAccountId] = useState<string | null>(null);
  const [bankAccountId, setBankAccountId] = useState<string | null>(null);


  // Hooks
  const companyId =
    userProfile?.currentCompanyId || userProfile?.companyIds?.[0] || "";


  const {
    combinedEvents,
    loading: combinedLoading,
    error: combinedError,
  } = useCombinedEvents(companyId);


  // Adaptive Navigation Logic
  useEffect(() => {
    // Only use URL sync on mobile
    if (!isMobile) return;


    const eventId = searchParams.get("eventId");
    if (eventId && combinedEvents.length > 0) {
      const event = combinedEvents.find((e) => e.id === eventId);
      if (event && event.type === "confirmed") {
        setSelectedEvent(event as BookingEvent);
        setIsEventDetailsOpen(true);
      }
    } else if (!eventId) {
      setIsEventDetailsOpen(false);
      setSelectedEvent(null);
    }
  }, [searchParams, combinedEvents, isMobile]);


  const handleEventDetailsChange = useCallback(
    (open: boolean) => {
      if (!open && isMobile) {
        // Clear eventId from URL only on mobile
        const params = new URLSearchParams(searchParams.toString());
        params.delete("eventId");
        router.replace(`?${params.toString()}`, { scroll: false });
      }
      setIsEventDetailsOpen(open);
    },
    [router, searchParams, isMobile]
  );


  const handleEventClick = (event: BookingEvent) => {
    if (isMobile) {
      // Use URL strategy on mobile
      const params = new URLSearchParams(searchParams.toString());
      params.set("eventId", event.id);
      router.push(`?${params.toString()}`, { scroll: false });
    } else {
      // Classic modal strategy on desktop
      setSelectedEvent(event);
      setIsEventDetailsOpen(true);
    }

      try {

    if (!cashAccountId && !bankAccountId) return combinedEvents;

    journalEntries.forEach((entry) => {
      if (entry.source !== "booking" || !entry.referenceId) return;


      let entryPaid = 0;
      entry.transactions.forEach((tx) => {
        if (
          tx.type === "debit" &&
          (tx.accountId === cashAccountId || tx.accountId === bankAccountId)
        ) {
          entryPaid += tx.amount;
        }
      });


      if (entryPaid > 0) {
        const current = paymentsMap.get(entry.referenceId) || 0;
        paymentsMap.set(entry.referenceId, current + entryPaid);
      }
    });


    return combinedEvents.map((event) => {
      const paid = paymentsMap.get(event.id) || 0;
      return {
        ...event,
        paidAmount: paid,
        balance: (event.amount || 0) - paid,

    },

  );

    setSelectedUnconfirmedEvent(event);
    setIsOnlineBookingDialogOpen(true);
  };


  const handleConfirmBooking = async (event: CombinedEvent) => {
    if (!companyId) return;
    try {
      const { data: newEvent, error } = await createBookingEvent(companyId, {
        title: event.title,
        startDate: event.startDate,
        endDate: event.endDate,
        customerName: event.customerName,
        customerEmail: event.customerEmail,
        customerPhone: event.customerPhone,
        place: event.place || "Online Booking",
        amount: event.amount || 0,
        balance: event.balance || 0,
        paidAmount: 0,
        status: "confirmed",
        source: "online_booking",
        description: event.description,
        currency: event.currency || "USD",
        baseCurrencyAmount: event.baseCurrencyAmount || event.amount || 0,
      });


      if (error) throw new Error(error);
      if (newEvent) {
        toast({ title: t("booking.confirmed_events") });
      }
    } catch (error) {
      console.error("Error confirming booking:", error);
      toast({
        title: t("booking.error_confirming_booking"),
        variant: "destructive",
      });
    }
  };

    console.log("Declining booking:", event);
    toast({ title: t("booking.booking_declined") });
  };

    setEditingEvent(event);
    handleEventDetailsChange(false);

      month: "long",
      year: "numeric",
    });

  if (combinedError || statsError)
    return <ErrorState error={combinedError || statsError} t={t} />;
  if (combinedLoading && !displayCompany) return <LoadingState t={t} />;


  return (
    <div dir={lang === "ar" ? "rtl" : "ltr"} className="flex flex-col h-full">

const useDebouncedValue = (value: string, delay: number = 300) => {
  const [debouncedValue, setDebouncedValue] = useState(value);


  useEffect(() => {
    const timer = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);


    return () => clearTimeout(timer);
  }, [value, delay]);


  return debouncedValue;
};

          const settings = getBookingSettings(currentCompany);
          const availableCurrencies =
            settings.currencies?.filter((c) => c.isActive) || [];
          setCurrencies(availableCurrencies);
          const base = availableCurrencies.find((c) => c.isBase);
          setBaseCurrency(base || null);
          if (base && !formData.currency && !editingEvent) {
            setFormData((prev) => ({ ...prev, currency: base.code }));
          }
        } catch (error) {
          console.error("Error loading currencies:", error);
          toast({
            title: t("booking.error_loading_currencies"),
            variant: "destructive",
          });
        }
      };
      loadCurrencies();
    }
  }, [currentCompany, open, formData.currency, editingEvent, t]);


  useEffect(() => {
    const fetchData = async () => {
      if (!companyId || !open) return;
      try {
        setDataLoaded(false);
        const [clientsRes, placesRes] = await Promise.allSettled([
          getClients(companyId, 50),
          getPlaces(companyId, 50),
        ]);


        if (clientsRes.status === "fulfilled" && clientsRes.value.data)
          setClients(clientsRes.value.data);
        if (placesRes.status === "fulfilled" && placesRes.value.data)
          setPlaces(placesRes.value.data);


        if (editingEvent) {
          if (clientsRes.status === "fulfilled" && clientsRes.value.data) {
            const matchingClient = clientsRes.value.data.find(
              (c) =>
                c.name.toLowerCase() === editingEvent.customerName.toLowerCase()
            );
            if (matchingClient) setSelectedClient(matchingClient);
          }
          if (placesRes.status === "fulfilled" && placesRes.value.data) {
            const matchingPlace = placesRes.value.data.find(
              (p) => p.name.toLowerCase() === editingEvent.place?.toLowerCase()
            );
            if (matchingPlace) setSelectedPlace(matchingPlace);
          }
        }
        setDataLoaded(true);
      } catch (error) {
        console.error("Error fetching data:", error);
        setDataLoaded(true);
      }
    };
    if (open) fetchData();
  }, [open, companyId, editingEvent]);


  useEffect(() => {
    if (debouncedSearchQuery.trim() === "") {
      setFilteredClients([]);
    } else {
      setFilteredClients(
        clients.filter((c) =>
          c.name.toLowerCase().includes(debouncedSearchQuery.toLowerCase())
        )
      );
    }
  }, [debouncedSearchQuery, clients]);


  useEffect(() => {
    if (debouncedPlaceSearchQuery.trim() === "") {
      setFilteredPlaces([]);
    } else {
      setFilteredPlaces(
        places.filter((p) =>
          p.name.toLowerCase().includes(debouncedPlaceSearchQuery.toLowerCase())
        )
      );
    }
  }, [debouncedPlaceSearchQuery, places]);


  const validateForm = () => {
    const errors: Record<string, string> = {};
    if (!formData.title.trim())
      errors.title = t("booking.validation.title_required");
    if (!formData.customerName.trim())
      errors.customerName = t("booking.validation.customer_name_required");
    if (!formData.fullAmount.trim())
      errors.fullAmount = t("booking.validation.amount_required");
    if (!formData.currency)
      errors.currency = t("booking.validation.currency_required");


    const fullAmount = parseFormattedNumber(formData.fullAmount);
    const paidAmount = parseFormattedNumber(formData.paidAmount || "0");
    if (isNaN(fullAmount) || fullAmount < 0)
      errors.fullAmount = t("booking.validation.invalid_amount");
    if (isNaN(paidAmount) || paidAmount < 0 || paidAmount > fullAmount)
      errors.paidAmount = t("booking.validation.invalid_paid_amount");


    if (paidAmount > 0 && !formData.method) {
      errors.method = t("booking.validation.payment_method_required");
    }


    phoneInputs.forEach((phone, index) => {
      const parsedPhone = parsePhoneNumber(phone);
      if (parsedPhone && !validatePhoneNumber(parsedPhone)) {
        errors[`phone_${index}`] = t("booking.validation.invalid_phone_format");
      }
    });


    if (
      formData.customerEmail &&
      !/\S+@\S+\.\S+/.test(formData.customerEmail)
    ) {
      errors.customerEmail = t("booking.validation.invalid_email");
    }


    setValidationErrors(errors);
    return Object.keys(errors).length === 0;
  };


  const handleInputChange = (field: string, value: string | Date) => {
    setFormData((prev) => ({ ...prev, [field]: value }));
    if (validationErrors[field]) {
      setValidationErrors((prev) => {
        const newErrors = { ...prev };
        delete newErrors[field];
        return newErrors;
      });
    }


    if (field === "customerName") {
      const nameValue = value as string;
      setSearchQuery(nameValue);
      setShowClientDropdown(true);
      if (nameValue === "" || !clients.some((c) => c.name === nameValue)) {
        setSelectedClient(null);
        if (nameValue !== selectedClient?.name) {
          setFormData((prev) => ({
            ...prev,
            customerEmail: "",
            customerPhone: "",
          }));
          setPhoneInputs([""]);
        }
      }
    }


    if (field === "place") {
      const placeValue = value as string;
      setPlaceSearchQuery(placeValue);
      setShowPlaceDropdown(true);
      if (placeValue === "" || !places.some((p) => p.name === placeValue)) {
        setSelectedPlace(null);
      }
    }
  };


  const handleAmountChange = (
    field: "fullAmount" | "paidAmount",
    value: string
  ) => {
    const formattedValue = formatNumberWithCommas(value);
    setFormData((prev) => ({ ...prev, [field]: formattedValue }));
  };


  const handleAddPhoneNumber = () => setPhoneInputs([...phoneInputs, ""]);
  const handleRemovePhoneNumber = (index: number) => {
    if (phoneInputs.length > 1) {
      const newInputs = phoneInputs.filter((_, i) => i !== index);
      setPhoneInputs(newInputs);
      if (index === 0 && newInputs.length > 0) {
        setFormData((prev) => ({ ...prev, customerPhone: newInputs[0] }));
      }
    }
  };


  const handlePhoneInputChange = (index: number, value: string) => {
    const formatted = formatPhoneNumber(value);
    const newInputs = [...phoneInputs];
    newInputs[index] = formatted;
    setPhoneInputs(newInputs);
    if (validationErrors[`phone_${index}`]) {
      setValidationErrors((prev) => {
        const newErrors = { ...prev };
        delete newErrors[`phone_${index}`];
        return newErrors;
      });
    }
    if (index === 0) {
      setFormData((prev) => ({ ...prev, customerPhone: formatted }));
    }
  };


  const handleClientSelect = (client: Client) => {
    setSelectedClient(client);
    const clientPhones = client.phones?.length
      ? client.phones
      : client.phone
      ? [client.phone]
      : [];
    const formattedPhones = clientPhones.length
      ? clientPhones.map(formatPhoneNumber)
      : [""];
    setPhoneInputs(formattedPhones);
    setFormData((prev) => ({
      ...prev,
      customerName: client.name,
      customerEmail: client.email || "",
      customerPhone: formattedPhones[0] || "",
    }));
    setSearchQuery("");
    setShowClientDropdown(false);
  };


  const handlePlaceSelect = (place: Place) => {
    setSelectedPlace(place);
    setFormData((prev) => ({ ...prev, place: place.name }));
    setPlaceSearchQuery("");
    setShowPlaceDropdown(false);
  };


  const getExchangeRate = (currencyCode: string) =>
    currencies.find((c) => c.code === currencyCode)?.rate || 1;
  const convertToBaseCurrency = (amount: number, currencyCode: string) =>
    amount / getExchangeRate(currencyCode);


  const getCurrentStatus = (): "confirmed" | "pending" | "partially_paid" => {
    const fullAmount = parseFormattedNumber(formData.fullAmount);
    const paidAmount = parseFormattedNumber(formData.paidAmount);
    if (paidAmount === 0) return "pending";
    if (paidAmount >= fullAmount) return "confirmed";
    return "partially_paid";
  };


  const getStatusColor = (status: string) => {
    switch (status) {
      case "confirmed":
        return "text-green-600";
      case "partially_paid":
        return "text-blue-600";
      case "pending":
        return "text-orange-600";
      default:
        return "text-muted-foreground";
    }
  };


  const getStatusDisplay = (status: string) => {
    switch (status) {
      case "confirmed":
        return t("booking.status.confirmed");
      case "partially_paid":
        return t("booking.status.partially_paid");
      case "pending":
        return t("booking.status.pending");
      default:
        return status;
    }
  };


  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (loading || !validateForm()) {
      toast({
        title: t("booking.validation.errors_found"),
        variant: "destructive",
      });
      return;
    }


    setLoading(true);
    try {
      const fullAmount = parseFormattedNumber(formData.fullAmount);
      const paidAmount = parseFormattedNumber(formData.paidAmount || "0");
      const baseCurrencyAmount = convertToBaseCurrency(
        fullAmount,
        formData.currency
      );


      let status: "confirmed" | "pending" | "partially_paid" = "pending";
      if (paidAmount >= fullAmount) status = "confirmed";
      else if (paidAmount > 0) status = "partially_paid";


      const uniqueEventPhones = [
        ...new Set(
          phoneInputs
            .map(parsePhoneNumber)
            .filter((p) => p && validatePhoneNumber(p))
        ),
      ];


      const eventData: Omit<BookingEvent, "id" | "updatedAt" | "paidAmount"> = {
        title: formData.title,
        customerName: formData.customerName,
        amount: fullAmount,
        balance: fullAmount - paidAmount,
        place: formData.place,
        status: status,
        startDate: formData.startDate,
        endDate: formData.startDate,
        createdAt: formData.createdAt,
        customerEmail: formData.customerEmail || undefined,
        customerPhones:
          uniqueEventPhones.length > 0 ? uniqueEventPhones : undefined,
        customerPhone: uniqueEventPhones[0] || undefined,
        currency: formData.currency,
        baseCurrencyAmount: baseCurrencyAmount,
        source: "manual",
        description: formData.description,
      };


      if (isEditMode && editingEvent) {
        const { error } = await updateBookingEvent(
          companyId,
          editingEvent.id,
          eventData
        );
        if (error) throw new Error(error);
        const updatedEvent: BookingEvent = {
          ...editingEvent,
          ...eventData,
          id: editingEvent.id,
          paidAmount,
          updatedAt: new Date(),
        };
        onEventUpdated(updatedEvent);
        toast({
          title: t("booking.event_updated"),
          description: t("booking.event_updated_successfully"),
        });
      } else {
        let clientPayload: (Partial<Client> & { name: string }) | undefined =
          undefined;
        if (formData.customerName.trim()) {
          clientPayload = {
            id: selectedClient?.id,
            name: formData.customerName.trim(),
            email: formData.customerEmail?.trim() || undefined,
            phones: uniqueEventPhones,
            phone: uniqueEventPhones[0],
          };
        }
        let placeDataForCreation: Omit<Place, "id"> | undefined = undefined;
        if (!selectedPlace && formData.place.trim()) {
          placeDataForCreation = { name: formData.place.trim(), address: "" };
        }


        const result = await createEventWithClientAndPlace(
          companyId,
          eventData,
          paidAmount,
          clientPayload,
          placeDataForCreation as any,
          formData.method
        );
        if (result.error || !result.data)
          throw new Error(result.error || "Failed to create event");


        onEventAdded(result.data);
        toast({
          title: t("booking.event_created"),
          description: t("booking.event_created_successfully"),
        });
      }


      resetForm();
      onOpenChange(false);
    } catch (error) {
      console.error("Error saving event:", error);
      toast({
        title: t("booking.error_saving_event"),
        description:
          error instanceof Error ? error.message : t("booking.unknown_error"),
        variant: "destructive",
      });
    } finally {
      setLoading(false);
    }
  };


  const resetForm = () => {
    setFormData({
      title: "",
      customerName: "",
      fullAmount: "",
      paidAmount: "0",
      place: "",
      startDate: initialStartDate || new Date(),
      createdAt: new Date(),
      customerEmail: "",
      customerPhone: "",
      currency: baseCurrency?.code || "",
      description: "",
      method: "",
    });
    setPhoneInputs([""]);
    setSelectedClient(null);
    setSelectedPlace(null);
    setSearchQuery("");
    setPlaceSearchQuery("");
    setShowClientDropdown(false);
    setShowPlaceDropdown(false);
    setIsEditMode(false);
    setValidationErrors({});
  };


  useEffect(() => {
    const handleClickOutside = (event: MouseEvent) => {
      if (
        clientContainerRef.current &&
        !clientContainerRef.current.contains(event.target as Node)
      )
        setShowClientDropdown(false);
      if (
        placeContainerRef.current &&
        !placeContainerRef.current.contains(event.target as Node)
      )
        setShowPlaceDropdown(false);
    };
    document.addEventListener("mousedown", handleClickOutside);
    return () => document.removeEventListener("mousedown", handleClickOutside);
  }, []);


  const currentStatus = getCurrentStatus();
  const fullAmountNum = parseFormattedNumber(formData.fullAmount);
  const paidAmountNum = parseFormattedNumber(formData.paidAmount);
  const balance = fullAmountNum - paidAmountNum;
  const selectedCurrency = currencies.find((c) => c.code === formData.currency);
  const currencySymbol = selectedCurrency?.symbol || "$";
  const dialogTitle = isEditMode
    ? t("booking.edit_event")
    : t("booking.add_new_event");
  const submitButtonText = isEditMode
    ? loading
      ? t("booking.updating")
      : t("booking.update_event")
    : loading
    ? t("booking.creating")
    : t("booking.create_event");


  return (
    <ResponsiveModal
      open={open}
      onOpenChange={onOpenChange}
      title={dialogTitle}
      description={isEditMode ? t("booking.edit_event_description") : t("booking.add_event_description")}
      className="lg:max-w-4xl"
    >
      <form
        onSubmit={handleSubmit}
        className="grid grid-cols-1 lg:grid-cols-2 lg:gap-x-8 gap-y-4"
      >
        {/* COL 1: Event & Client Details */}
        <div className="flex flex-col gap-y-4">
          <div className="grid sm:grid-cols-2 gap-4">
            <div className="space-y-2">
              <Label htmlFor="customerName" className="flex items-center gap-2">
                <User className="h-4 w-4" />
                {t("booking.customer_name")} *
              </Label>
              <div className="relative" ref={clientContainerRef}>
                <Input
                  id="customerName"
                  placeholder={t("booking.customer_name_placeholder")}
                  value={formData.customerName}
                  onChange={(e) =>
                    handleInputChange("customerName", e.target.value)
                  }
                  onFocus={() => setShowClientDropdown(true)}
                  required
                  className={validationErrors.customerName ? "border-red-500" : ""}
                />
                {!isEditMode && filteredClients.length > 0 && showClientDropdown && (
                  <div className="absolute z-50 w-full mt-1 bg-popover border border-border rounded-md shadow-lg max-h-60 overflow-auto">
                    {filteredClients.map((client) => (
                      <div
                        key={client.id}
                        className="px-3 py-2 cursor-pointer hover:bg-accent border-b last:border-b-0"
                        onClick={() => handleClientSelect(client)}
                      >
                        <div className="font-medium">{client.name}</div>
                        {client.email && (
                          <div className="text-sm text-muted-foreground">{client.email}</div>
                        )}
                      </div>
                    ))}
                  </div>
                )}
              </div>
              {selectedClient && !isEditMode && (
                <p className="text-sm text-green-600">{t("booking.existing_client_selected")}</p>
              )}
              {validationErrors.customerName && (
                <p className="text-sm text-red-500 flex items-center gap-1">
                  <AlertCircle className="h-3 w-3" />
                  {validationErrors.customerName}
                </p>
              )}
            </div>


            <div className="space-y-2">
              <Label htmlFor="title" className="flex items-center gap-2">
                <CalendarIcon className="h-4 w-4" />
                {t("booking.event_title")} *
              </Label>
              <Input
                id="title"
                placeholder={t("booking.event_title_placeholder")}
                value={formData.title}
                onChange={(e) => handleInputChange("title", e.target.value)}
                required
                className={validationErrors.title ? "border-red-500" : ""}
              />
              {validationErrors.title && (
                <p className="text-sm text-red-500 flex items-center gap-1">
                  <AlertCircle className="h-3 w-3" />
                  {validationErrors.title}
                </p>
              )}
            </div>
          </div>


          {/* Phone Numbers */}
          <div className="space-y-3">
            <div className="flex items-center justify-between">
              <Label className="flex items-center gap-2">
                <Phone className="h-4 w-4" />
                {t("booking.customer_phone")}
              </Label>
              <Button
                type="button"
                variant="outline"
                size="sm"
                onClick={handleAddPhoneNumber}
                className="h-7 text-xs"
              >
                <Plus className="h-3 w-3 mr-1" />
                {t("booking.add_another_phone")}
              </Button>
            </div>
            <div className="space-y-2">
              {phoneInputs.map((phone, index) => (
                <div key={index} className="flex gap-2 items-center">
                  <Input
                    type="tel"
                    placeholder="091-234-5678"
                    value={phone}
                    onChange={(e) => handlePhoneInputChange(index, e.target.value)}
                    maxLength={12}
                    className={validationErrors[`phone_${index}`] ? "border-red-500" : ""}
                  />
                  {phoneInputs.length > 1 && (
                    <Button
                      type="button"
                      variant="destructive"
                      size="icon"
                      onClick={() => handleRemovePhoneNumber(index)}
                      title={t("booking.remove_phone")}
                    >
                      <Trash2 className="h-4 w-4" />
                    </Button>
                  )}
                </div>
              ))}
            </div>
          </div>


          <div className="grid sm:grid-cols-2 gap-4">
            <div className="space-y-2">
              <Label htmlFor="customerEmail">{t("booking.customer_email")}</Label>
              <Input
                id="customerEmail"
                type="email"
                placeholder="email@example.com"
                value={formData.customerEmail}
                onChange={(e) => handleInputChange("customerEmail", e.target.value)}
                className={validationErrors.customerEmail ? "border-red-500" : ""}
              />
            </div>
            <div className="space-y-2">
              <Label htmlFor="place" className="flex items-center gap-2">
                <MapPin className="h-4 w-4" />
                {t("booking.place")}
              </Label>
              <div className="relative" ref={placeContainerRef}>
                <Input
                  id="place"
                  placeholder={t("booking.place_placeholder")}
                  value={formData.place}
                  onChange={(e) => handleInputChange("place", e.target.value)}
                  onFocus={() => setShowPlaceDropdown(true)}
                />
              </div>
            </div>
          </div>


          <div className="grid sm:grid-cols-2 gap-4">
            {/* Start Date */}
            <div className="space-y-2">
              <Label htmlFor="startDate" className="flex items-center gap-2">
                <CalendarIcon className="h-4 w-4" />
                {t("booking.start_date")}
              </Label>
              <Popover>
                <PopoverTrigger asChild>
                  <Button
                    variant={"outline"}
                    className={cn(
                      "w-full justify-start text-left font-normal",
                      !formData.startDate && "text-muted-foreground"
                    )}
                  >
                    <CalendarIcon className="h-4 w-4 mr-2" />
                    {formData.startDate ? format(formData.startDate, "PPP") : <span>Pick a date</span>}
                  </Button>
                </PopoverTrigger>
                <PopoverContent className="w-auto p-0">
                  <Calendar
                    mode="single"
                    selected={formData.startDate}
                    onSelect={(date) => date && handleInputChange("startDate", date)}
                    initialFocus
                  />
                </PopoverContent>
              </Popover>
            </div>
            {/* Creation Date */}
            <div className="space-y-2">
              <Label htmlFor="createdAt" className="flex items-center gap-2">
                <CalendarIcon className="h-4 w-4" />
                {t("booking.creation_date")}
              </Label>
              <Popover>
                <PopoverTrigger asChild>
                  <Button
                    variant={"outline"}
                    className={cn(
                      "w-full justify-start text-left font-normal",
                      !formData.createdAt && "text-muted-foreground"
                    )}
                  >
                    <CalendarIcon className="h-4 w-4 mr-2" />
                    {formData.createdAt ? format(formData.createdAt, "PPP") : <span>Pick a date</span>}
                  </Button>
                </PopoverTrigger>
                <PopoverContent className="w-auto p-0">
                  <Calendar
                    mode="single"
                    selected={formData.createdAt}
                    onSelect={(date) => date && handleInputChange("createdAt", date)}
                    initialFocus
                  />
                </PopoverContent>
              </Popover>
            </div>
          </div>


          <div className="space-y-2">
            <Label>{t("booking.status")}</Label>
            <div className="p-3 border rounded-md bg-muted/30">
              <div className="flex items-center justify-between">
                <span className={cn("font-medium", getStatusColor(currentStatus))}>
                  {getStatusDisplay(currentStatus)}
                </span>
                <div className="text-right">
                  <div className={`text-sm font-bold ${balance <= 0 ? "text-green-600" : "text-red-600"}`}>
                    {t("booking.due_amount")}: {currencySymbol}{balance.toLocaleString()}
                  </div>
                </div>
              </div>
            </div>
          </div>
        </div>


        {/* COL 2: Financials & Actions */}
        <div className="flex flex-col justify-between">
          <div className="space-y-2">
            <Label className="flex items-center gap-2">
              <Wallet className="h-4 w-4" />
              {t("booking.financials")}
            </Label>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-3 p-4 border rounded-lg bg-muted/30">
              <div className="space-y-1">
                <Label htmlFor="fullAmount" className="text-xs">{t("booking.full_amount")} *</Label>
                <Input
                  id="fullAmount"
                  value={formData.fullAmount}
                  onChange={(e) => handleAmountChange("fullAmount", e.target.value)}
                  required
                />
              </div>
              <div className="space-y-1">
                <Label htmlFor="currency" className="text-xs">{t("booking.currency")} *</Label>
                <Select
                  value={formData.currency}
                  onValueChange={(value) => handleInputChange("currency", value)}
                  required
                >
                  <SelectTrigger><SelectValue /></SelectTrigger>
                  <SelectContent>
                    {currencies.map((c) => (
                      <SelectItem key={c.code} value={c.code}>{c.code}</SelectItem>
                    ))}
                  </SelectContent>
                </Select>
              </div>
              <div className="space-y-1">
                <Label htmlFor="paidAmount" className="text-xs">{t("booking.paid_amount")}</Label>
                <Input
                  id="paidAmount"
                  value={formData.paidAmount}
                  onChange={(e) => handleAmountChange("paidAmount", e.target.value)}
                />
              </div>
              <div className="space-y-1">
                <Label htmlFor="method" className="text-xs">{t("booking.payment_method")} *</Label>
                <Select
                  value={formData.method}
                  onValueChange={(value) => handleInputChange("method", value)}
                >
                  <SelectTrigger><SelectValue /></SelectTrigger>
                  <SelectContent>
                    {paymentMethods.map((m) => (
                      <SelectItem key={m.value} value={m.value}>{m.label}</SelectItem>
                    ))}
                  </SelectContent>
                </Select>
              </div>
            </div>
          </div>


          <div className={`flex flex-col sm:flex-row gap-3 pt-4 ${lang === "ar" ? "sm:flex-row-reverse" : ""}`}>
            <Button
              type="button"
              variant="outline"
              onClick={() => { resetForm(); onOpenChange(false); }}
              disabled={loading}
              className="sm:flex-1"
            >
              {t("cancel")}
            </Button>
            <Button
              type="submit"
              className="bg-linear-to-r from-primary to-secondary hover:opacity-90 sm:flex-1"
              disabled={loading}
            >
              {loading && <Loader2 className="h-4 w-4 mr-2 animate-spin" />}
              {submitButtonText}
            </Button>
          </div>
        </div>
      </form>
    </ResponsiveModal>
  );
}

import type { CombinedEvent, BookingEvent } from "@/types/booking";


interface CalendarViewProps {
  t: any;
  lang: string;
  currentDate: Date;
  selectedDate: Date | undefined;
  combinedEvents: CombinedEvent[];
  navigateMonth: (direction: "prev" | "next") => void;
  goToToday: () => void;
  formatMonthYear: (date: Date) => string;
  formatDate: (date: Date, options?: any) => string;
  dateHasEvents: (date: Date) => boolean;
  getEventTypesForDate: (date: Date) => {
    hasConfirmed: boolean;
    hasUnconfirmed: boolean;
  };
  setSelectedDate: (date: Date) => void;
  handleDayDoubleClick: (date: Date) => void;
  selectedDateEvents: CombinedEvent[];
  handleEventClick: (event: BookingEvent) => void;
  handleUnconfirmedEventClick: (event: CombinedEvent) => void;
}


export function CalendarView({
  t,
  lang,
  currentDate,
  selectedDate,
  combinedEvents,
  navigateMonth,
  goToToday,
  formatMonthYear,
  formatDate,
  dateHasEvents,
  getEventTypesForDate,
  setSelectedDate,
  handleDayDoubleClick,
  selectedDateEvents,
  handleEventClick,
  handleUnconfirmedEventClick,
}: CalendarViewProps) {
  const weekdays = [
    t("booking.weekdays.sun"),
    t("booking.weekdays.mon"),
    t("booking.weekdays.tue"),
    t("booking.weekdays.wed"),
    t("booking.weekdays.thu"),
    t("booking.weekdays.fri"),
    t("booking.weekdays.sat"),
  ];


  return (
    <Card className="lg:col-span-2">
      <CardHeader className="px-3 sm:px-6">
        <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center gap-3">
          <CardTitle className="flex items-center gap-2 text-base sm:text-lg">
            <Calendar className="h-4 w-4 sm:h-5 sm:w-5 text-primary" />
            {t("booking.calendar_view")}
          </CardTitle>
          <div className="flex flex-wrap items-center gap-2 w-full sm:w-auto">
            <div className="flex items-center gap-1">
              <Button
                variant="outline"
                size="sm"
                onClick={() => navigateMonth("prev")}
                className="h-8 w-8 p-0"
              >
                {lang === "ar" ? (
                  <ChevronRight className="h-3 w-3 sm:h-4 sm:w-4" />
                ) : (
                  <ChevronLeft className="h-3 w-3 sm:h-4 sm:w-4" />
                )}
              </Button>
              <span className="text-xs sm:text-sm font-medium min-w-[120px] text-center">
                {formatMonthYear(currentDate)}
              </span>
              <Button
                variant="outline"
                size="sm"
                onClick={() => navigateMonth("next")}
                className="h-8 w-8 p-0"
              >
                {lang === "ar" ? (
                  <ChevronLeft className="h-3 w-3 sm:h-4 sm:w-4" />
                ) : (
                  <ChevronRight className="h-3 w-3 sm:h-4 sm:w-4" />
                )}
              </Button>
            </div>
            <Button
              variant="outline"
              size="sm"
              onClick={goToToday}
              className="h-8 px-2 text-xs"
            >
              <Home className="h-3 w-3 mr-1 sm:mr-2" />
              <span className="hidden sm:inline">
                {t("booking.today") || "Today"}
              </span>
            </Button>
          </div>
        </div>
        <CardDescription className="text-xs sm:text-sm">
          {t("booking.calendar_description")}
          <span className="block mt-1 text-xs text-blue-600">
            {t("booking.double_click_instruction")}
          </span>
        </CardDescription>
      </CardHeader>
      <CardContent className="px-3 sm:px-6">
        <div className="min-h-[350px] sm:min-h-[400px] border rounded-lg bg-muted/20 p-2 sm:p-4 overflow-hidden">
          {/* Simple Calendar Implementation */}
          <div className="grid grid-cols-7 gap-0.5 sm:gap-1 mb-3 sm:mb-4">
            {weekdays.map((day, index) => (
              <div
                key={day}
                className="text-center text-xs sm:text-sm font-medium py-1 sm:py-2 text-foreground truncate"
              >
                {lang === "ar"
                  ? // Standard Arabic Calendar Abbreviations: Su=ح, Mo=ن, Tu=ث, We=ر, Th=خ, Fr=ج, Sa=س
                    ["ح", "ن", "ث", "ر", "خ", "ج", "س"][index]
                  : day.substring(0, 1)}
              </div>
            ))}
            {Array.from({ length: 35 }, (_, i) => {
              const firstDayOfMonth = new Date(
                currentDate.getFullYear(),
                currentDate.getMonth(),
                1
              );
              const firstDayOfWeek = firstDayOfMonth.getDay();
              const date = new Date(
                currentDate.getFullYear(),
                currentDate.getMonth(),
                1 - firstDayOfWeek + i
              );
              const isCurrentMonth = date.getMonth() === currentDate.getMonth();
              const isToday = date.toDateString() === new Date().toDateString();
              const isSelected =
                selectedDate?.toDateString() === date.toDateString();
              const hasEvent = dateHasEvents(date);
              const { hasConfirmed, hasUnconfirmed } =
                getEventTypesForDate(date);


              let eventBackground = "transparent";
              if (hasConfirmed && hasUnconfirmed) {
                eventBackground =
                  "linear-gradient(135deg,#3bf6abff 50%, #ef4444 50%)";
              } else if (hasConfirmed) {
                eventBackground = "#3bf6abff";
              } else if (hasUnconfirmed) {
                eventBackground = "#ef4444";
              }


              return (
                <button
                  key={i}
                  onClick={() => setSelectedDate(date)}
                  onDoubleClick={() => handleDayDoubleClick(date)}
                  className={`
                    relative h-8 sm:h-10 md:h-12 rounded-md sm:rounded-lg transition-all duration-200
                    ${
                      isSelected
                        ? "bg-blue-500 text-white border-blue-900 shadow-md sm:shadow-lg"
                        : isToday
                        ? "bg-blue-50 border-blue-200 dark:bg-blue-950 dark:border-blue-950"
                        : "border-gray-200 dark:border-gray-700 hover:border-blue-300 hover:bg-blue-50 dark:hover:bg-blue-950/50"
                    }
                    ${!isCurrentMonth ? "text-muted-foreground/40" : ""}
                    flex items-center justify-center cursor-pointer select-none group
                    w-full min-w-0
                  `}
                  title={t("booking.double_click_to_add_event")}
                >
                  {hasEvent && !isSelected && (
                    <div
                      className="absolute inset-0.5 sm:inset-1 rounded-sm sm:rounded-md transition-opacity duration-200"
                      style={{ background: eventBackground }}
                    />
                  )}
                  <span
                    className={`
                    text-xs sm:text-sm font-medium z-10
                    ${
                      isSelected
                        ? "text-white"
                        : hasEvent && !isSelected
                        ? "text-white"
                        : isToday
                        ? "text-blue-700 dark:text-blue-300"
                        : "text-foreground"
                    }
                  `}
                  >
                    {date.getDate()}
                  </span>
                </button>
              );
            })}
          </div>


          {/* Legend */}
          <div className="flex flex-row sm:flex-row justify-center items-start sm:items-center gap-2 sm:gap-4 mt-4 text-xs text-muted-foreground">
            <div className="flex items-center gap-2">
              <div className="w-3 h-3 sm:w-4 sm:h-4 rounded-md bg-green-500 shadow-sm"></div>
              <span className="whitespace-nowrap">
                {t("booking.confirmed_events")}
              </span>
            </div>
            <div className="flex items-center gap-2">
              <div className="w-3 h-3 sm:w-4 sm:h-4 rounded-md bg-red-500 shadow-sm"></div>
              <span className="whitespace-nowrap">
                {t("booking.unconfirmed_events")}
              </span>
            </div>


            <div className="flex items-center gap-2">
              <div className="w-3 h-3 sm:w-4 sm:h-4 rounded-md bg-blue-500"></div>
              <span className="whitespace-nowrap">{t("booking.selected")}</span>
            </div>
          </div>


          {/* Selected Date Events */}
          {selectedDate && (
            <div className="mt-4 sm:mt-6">
              <h3 className="font-semibold mb-2 sm:mb-3 text-sm sm:text-base text-foreground">
                {t("booking.events_for")} {formatDate(selectedDate)}
              </h3>
              {selectedDateEvents.length > 0 ? (
                <div className="space-y-2">
                  {selectedDateEvents.map((event) => (
                        <div
                          key={event.id}
                          className="flex flex-col sm:flex-row sm:justify-between sm:items-center p-2 sm:p-3 border rounded-lg bg-background cursor-pointer hover:bg-accent/50 transition-colors"
                          onClick={() => event.type === 'confirmed' ? handleEventClick(event as BookingEvent) : handleUnconfirmedEventClick(event)}
                        >
                          <div className="mb-1 sm:mb-0">
                            <p className="font-medium text-sm sm:text-base text-foreground truncate">
                              {event.type === 'confirmed' ? event.customerName : event.title}
                            </p>
                            <p className="text-xs sm:text-sm text-muted-foreground truncate">
                              {event.title} • {event.place} •{" "}
                              {event.startDate.toLocaleTimeString([], {
                                hour: "2-digit",
                                minute: "2-digit",
                              })}
                            </p>
                          </div>
                          <Badge
                            variant={event.type === 'confirmed' ? (event.status === "confirmed"
                                ? "default"
                                : event.status === "pending"
                                ? "secondary"
                                : "destructive") : 'outline'}
                            className={`text-xs w-fit mt-1 sm:mt-0 ${event.type === 'unconfirmed' && 'bg-yellow-100 text-yellow-800'}`}>
                            {event.type === 'confirmed' ? t(`booking.status.${event.status}`) : t("booking.pending_confirmation")}
                          </Badge>
                        </div>
                      )
                  )}
                </div>
              ) : (
                <p className="text-muted-foreground text-center py-3 sm:py-4 text-sm sm:text-base">
                  {t("booking.no_events_scheduled")}
                </p>
              )}
            </div>
          )}
        </div>
      </CardContent>
    </Card>
  );
}

interface EventDetailsDialogProps {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  event: BookingEvent | null;
  onEdit: (event: BookingEvent) => void;
  onDelete: (eventId: string) => void;
  onEventUpdated?: () => void;
}


export function EventDetailsDialog({
  open,
  onOpenChange,
  event,
  onEdit,
  onDelete,
  onEventUpdated,
}: EventDetailsDialogProps) {
  const { t, lang } = useLang();
  const { userProfile } = useAuth();
  const { toast } = useToast();
  const [totalPaid, setTotalPaid] = useState(0);
  const [loadingFinancials, setLoadingFinancials] = useState(true);
  const [activeAccordion, setActiveAccordion] = useState<string>("");
  const [isRescheduleOpen, setIsRescheduleOpen] = useState(false);
  const [localEvent, setLocalEvent] = useState<BookingEvent | null>(null);
  const [whatsAppDialogOpen, setWhatsAppDialogOpen] = useState(false);
  const [selectedPhone, setSelectedPhone] = useState<string | null>(null);
  const [useTemplateMessage, setUseTemplateMessage] = useState(false);


  const [confirmationDialog, setConfirmationDialog] = useState<{
    open: boolean;
    title: string;
    description: string;
    onConfirm: () => void;
    variant?: "default" | "destructive";
  }>({
    open: false,
    title: "",
    description: "",
    onConfirm: () => {},
  });


  useEffect(() => {
    if (event) {
      setLocalEvent(event);
    }
  }, [event]);


  useEffect(() => {
    if (!open || !localEvent || !userProfile) return;


    setLoadingFinancials(true);
    const companyId =
      userProfile.currentCompanyId || userProfile.companyIds?.[0];
    if (!companyId) {
      setLoadingFinancials(false);
      return;
    }


    let unsubscribe: (() => void) | undefined;


    const getPaidAmount = async () => {
      try {
        const cashAccountRes = await getAccountByName(companyId, "Cash");
        const cashAccountId = cashAccountRes.data?.id;


        const bankAccountRes = await getAccountByName(
          companyId,
          "Bank Transfers"
        );
        const bankAccountId = bankAccountRes.data?.id;


        if (!cashAccountId && !bankAccountId) {
          console.warn(
            "Neither Cash nor Bank Transfers account found for paid amount calculation."
          );
          setTotalPaid(0);
          setLoadingFinancials(false);
          return;
        }


        const q = query(
          collection(db, "companies", companyId, "journalEntries"),
          where("source", "==", "booking"),
          where("referenceId", "==", localEvent.id)
        );


        unsubscribe = onSnapshot(q, (snapshot) => {
          let paid = 0;
          snapshot.forEach((doc) => {
            const entry = doc.data() as JournalEntry;
            entry.transactions?.forEach((transaction: any) => {
              if (
                transaction.type === "debit" &&
                (transaction.accountId === cashAccountId ||
                  transaction.accountId === bankAccountId)
              ) {
                paid += transaction.amount || 0;
              }
            });
          });
          setTotalPaid(paid);
          setLoadingFinancials(false);
        });
      } catch (error) {
        console.error("Error fetching total paid amount:", error);
        setLoadingFinancials(false);
      }
    };


    getPaidAmount();


    return () => {
      if (unsubscribe) {
        unsubscribe();
      }
    };
  }, [open, localEvent, userProfile]);


  if (!localEvent) return null;


  const balance = localEvent.amount - totalPaid;
  const progressPercentage =
    localEvent.amount > 0 ? (totalPaid / localEvent.amount) * 100 : 0;


  const formatCurrency = (amount: number, currency: string = "USD") => {
    try {
      return new Intl.NumberFormat("en-US", {
        style: "currency",
        currency,
        minimumFractionDigits: 2,
        maximumFractionDigits: 2,
      }).format(amount);
    } catch (e) {
      return `${amount.toFixed(2)} ${currency}`;
    }
  };


  const openWhatsApp = () => {
    if (!selectedPhone || !localEvent) return;
    const phone = selectedPhone.replace(/\D/g, "");
    const templateMessage = `عميلنا الكريم ${
      localEvent.customerName
    } السلام عليكم  \nنرجو منكم التكرم بإكمال باقي المبلغ ${formatCurrency(
      balance,
      localEvent.currency
    )}  \nشاكرين لكم حسن التعاون  \nنتمنى لكم يوما سعيدا`;
    const url = useTemplateMessage
      ? `https://wa.me/${phone}?text=${encodeURIComponent(templateMessage)}`
      : `https://wa.me/${phone}`;
    window.open(url, "_blank");
    setWhatsAppDialogOpen(false);
  };


  const formatDate = (
    date?: Date | Timestamp,
    options: Intl.DateTimeFormatOptions = {}
  ) => {
    if (!date) return "N/A";
    const dateObj = date instanceof Timestamp ? date.toDate() : date;
    if (isNaN(dateObj.getTime())) return "N/A";
    return dateObj.toLocaleDateString("en-US", {
      year: "numeric",
      month: "long",
      day: "numeric",
      weekday: "long",
      ...options,
    });
  };


  const formatTime = (date?: Date | Timestamp) => {
    if (!date) return "N/A";
    const dateObj = date instanceof Timestamp ? date.toDate() : date;
    if (isNaN(dateObj.getTime())) return "N/A";
    return dateObj.toLocaleTimeString("en-US", {
      hour: "2-digit",
      minute: "2-digit",
    });
  };


  const isPlaceholderDate = (date?: Date | Timestamp) => {
    if (!date) return true;
    const dateObj = date instanceof Timestamp ? date.toDate() : date;
    return dateObj.getTime() === 0;
  };


  const getStatusConfig = (status: BookingEvent["status"]) => {
    const config = {
      confirmed: {
        variant: "default" as const,
        icon: CheckCircle2,
        className: "bg-green-100 text-green-800 border-green-200",
      },
      postponed: {
        variant: "secondary" as const,
        icon: CalendarClock,
        className: "bg-orange-100 text-orange-800 border-orange-200",
      },
      pending: {
        variant: "secondary" as const,
        icon: Clock,
        className: "bg-yellow-100 text-yellow-800 border-yellow-200",
      },
      cancelled: {
        variant: "destructive" as const,
        icon: XCircle,
        className: "bg-red-100 text-red-800 border-red-200",
      },
      partially_paid: {
        variant: "secondary" as const,
        icon: AlertCircle,
        className: "bg-blue-100 text-blue-800 border-blue-200",
      },
    };
    return (
      config[status] || {
        variant: "secondary" as const,
        icon: Clock,
        className: "bg-gray-100 text-gray-800 border-gray-200",
      }
    );
  };


  const statusConfig = getStatusConfig(localEvent.status);
  const StatusIcon = statusConfig.icon;


  const handlePaymentAdded = async (paymentData: Payment) => {
    toast({
      title: t("booking.payment_success"),
      description: t("booking.payment_add_success"),
    });
    if (onEventUpdated) onEventUpdated();
    setActiveAccordion(""); // Close accordion
  };


  const handlePaymentUpdated = () => {
    toast({
      title: t("booking.payment_updated_success"),
      description: t("booking.payment_updated_success_description"),
    });
    if (onEventUpdated) onEventUpdated();
  };


  const confirmAction = (
    title: string,
    description: string,
    onConfirm: () => void,
    variant: "default" | "destructive" = "default"
  ) => {
    setConfirmationDialog({
      open: true,
      title,
      description,
      onConfirm,
      variant,
    });
  };


  const handlePostponeEvent = () => {
    if (!userProfile?.companyIds?.length || !localEvent) return;
    confirmAction(
      t("booking.confirm_postpone"),
      t("booking.confirm_postpone_message"),
      async () => {
        const companyId =
          userProfile.currentCompanyId || userProfile.companyIds[0];
        const placeholderDate = new Date(0);
        const updateData = {
          status: "postponed" as const,
          startDate: placeholderDate,
          endDate: placeholderDate,
          updatedAt: new Date(),
        };
        const { error } = await updateBookingEvent(
          companyId,
          localEvent.id,
          updateData
        );
        if (error) {
          toast({
            title: t("booking.error"),
            description: t("booking.event_postponed_error"),
            variant: "destructive",
          });
        } else {
          setLocalEvent({ ...localEvent, ...updateData });
          if (onEventUpdated) onEventUpdated();
          toast({
            title: t("booking.success"),
            description: t("booking.event_postponed_success"),
          });
        }
      }
    );
  };


  const handleRescheduleConfirm = async (
    newStartDate: Date,
    newEndDate: Date
  ) => {
    if (!userProfile?.companyIds?.length || !localEvent) return;
    const companyId = userProfile.currentCompanyId || userProfile.companyIds[0];
    let newStatus: BookingEvent["status"] =
      totalPaid >= localEvent.amount
        ? "confirmed"
        : totalPaid > 0
        ? "partially_paid"
        : "pending";
    const updateData = {
      status: newStatus,
      startDate: newStartDate,
      endDate: newEndDate,
      updatedAt: new Date(),
    };
    const { error } = await updateBookingEvent(
      companyId,
      localEvent.id,
      updateData
    );
    if (error) {
      toast({
        title: t("booking.error"),
        description: t("booking.event_rescheduled_error"),
        variant: "destructive",
      });
    } else {
      setLocalEvent({ ...localEvent, ...updateData });
      if (onEventUpdated) onEventUpdated();
      toast({
        title: t("booking.success"),
        description: t("booking.event_rescheduled_success"),
      });
      setIsRescheduleOpen(false);
    }
  };


  const handleDeleteClick = () => {
    confirmAction(
      t("booking.confirm_delete"),
      t("booking.confirm_delete_message"),
      async () => {
        await onDelete(localEvent.id);
        setConfirmationDialog((prev) => ({ ...prev, open: false }));
        onOpenChange(false);
      },
      "destructive"
    );
  };


  return (
    <>
      <ResponsiveModal
        open={open}
        onOpenChange={onOpenChange}
        title={localEvent.title}
        description={t("booking.event_details_description")}
        className="md:max-w-[1000px]"
      >
        <div className="grid grid-cols-1 md:grid-cols-2 gap-6 items-start">
          {/* Left Column: Event Details */}
          <div className="space-y-6">
            <Card className="h-full">
              <CardHeader className="pb-3">
                <CardTitle className="text-base sm:text-lg flex items-center gap-2">
                  <span className="truncate">{localEvent.title}</span>
                  <Badge
                    variant={statusConfig.variant}
                    className={`${statusConfig.className} w-fit mt-1 sm:mt-0`}
                  >
                    <StatusIcon className="h-3 w-3 mr-1" />
                    {t(`booking.status.${localEvent.status}`)}
                  </Badge>
                </CardTitle>
              </CardHeader>
              <CardContent className="space-y-4">
                <div className="space-y-4">
                  {/* Customer Info */}
                  <div className="space-y-3">
                    <div className="flex items-center gap-2">
                      <User className="h-4 w-4 text-muted-foreground shrink-0" />
                      <span className="font-medium text-sm sm:text-base">
                        {t("booking.customer")}:
                      </span>
                    </div>
                    <div className="ml-6 space-y-1">
                      <p className="text-sm wrap-break-word font-semibold">
                        {localEvent.customerName}
                      </p>
                      {(localEvent as any).customerEmail && (
                        <p className="text-sm text-muted-foreground wrap-break-word">
                          {(localEvent as any).customerEmail}
                        </p>
                      )}
                      {Array.isArray(localEvent.customerPhones) &&
                      localEvent.customerPhones.length > 0 ? (
                        <div className="space-y-2">
                          {localEvent.customerPhones.map((phone, index) => (
                            <a
                              key={index}

              entry.transactions?.forEach((transaction: any) => {
                if (transaction.type === "debit") {
                    let method = "";
                    if (transaction.accountId === cashAccountId) {
                        method = "Cash";
                    } else if (transaction.accountId === bankAccountId) {
                        method = "Bank Transfer";
                    }


                    if (method) {
                        fetchedPayments.push({
                            id: doc.id,
                            amount: transaction.amount,
                            date: entry.date.toDate(),
                            method: method,
                            notes: entry.description || "",
                        });
                    }
                }
              });
            });
            setPayments(
              fetchedPayments.sort(
                (a, b) => b.date.getTime() - a.date.getTime()
              )
            );
            setLoading(false);
          },
          (error) => {
            console.error("Error fetching payment journal entries:", error);
            setLoading(false);
          }
        );
        return unsubscribe;
      } catch (error) {
        console.error("Error setting up payment fetch:", error);
        setLoading(false);
      }
    };


    const unsubscribePromise = fetchPayments();


    return () => {
      unsubscribePromise.then((unsubscribe) => {
        if (unsubscribe) unsubscribe();
      });
    };
  }, [event, companyId]);


  const handleEditPayment = (payment: FormattedPayment) => {
    setEditingPayment(payment);
    setIsAddingPayment(false);
  };


  const handleDeletePayment = async (payment: FormattedPayment) => {
    if (!companyId) return;


    try {
      const batch = writeBatch(db);


      // 1. Delete the journal entry
      const journalEntryRef = doc(
        db,
        "companies",
        companyId,
        "journalEntries",
        payment.id
      );
      batch.delete(journalEntryRef);


      // 2. Revert the balance on the booking event
      const bookingEventRef = doc(
        db,
        "companies",
        companyId,
        "bookingEvents",
        event.id
      );
      const newPaidAmount = (event.paidAmount || 0) - payment.amount;
      const newBalance = event.amount - newPaidAmount;
      batch.update(bookingEventRef, {
        paidAmount: newPaidAmount,
        balance: newBalance,
      });


      await batch.commit();


      toast({ title: t("booking.payment_deleted_success") });
      if (onPaymentUpdated) onPaymentUpdated();
    } catch (error) {
      console.error("Error deleting payment:", error);
      toast({
        title: t("booking.error"),
        description: t("booking.payment_deleted_error"),
        variant: "destructive",
      });
    }
  };


  const onPaymentAddedOrUpdated = () => {
    setEditingPayment(null);
    setIsAddingPayment(false);
    if (onPaymentUpdated) onPaymentUpdated();
  };


  const getPaymentMethodIcon = (method: string) => {
    const lowerMethod = method.toLowerCase();
    if (lowerMethod.includes("card")) return CreditCard;
    if (lowerMethod.includes("bank") || lowerMethod.includes("transfer"))
      return Banknote;
    if (lowerMethod.includes("cash")) return Wallet;
    return DollarSign;
  };


  const statusConfig = {
    icon: CheckCircle2,
    color: "text-primary",
    bgColor: "bg-primary/10",
  };
  const StatusIcon = statusConfig.icon;


  const formatCurrency = (amount: number, currencyCode: string) => {
    try {
      return new Intl.NumberFormat("en-US", {
        style: "currency",
        currency: currencyCode,
        minimumFractionDigits: 2,
        maximumFractionDigits: 2,
      }).format(amount);
    } catch (e) {
      return `${amount.toFixed(2)} ${currencyCode}`;
    }
  };


  const toggleReceipt = (paymentId: string) => {
    setOpenReceipts((prev) => {
      const newSet = new Set(prev);
      if (newSet.has(paymentId)) {
        newSet.delete(paymentId);
      } else {
        newSet.add(paymentId);
      }
      return newSet;
    });
  };


  if (loading) {
    return (
      <Card className="border-dashed">
        <CardContent className="flex flex-col items-center justify-center py-10 text-center">
          <Loader2 className="h-12 w-12 text-muted-foreground/30 mb-4 animate-spin" />
          <p className="text-muted-foreground font-semibold">{t("loading")}</p>
        </CardContent>
      </Card>
    );
  }


  return (
    <div className="space-y-4">
      {/* Payment Form for Adding/Editing */}
      {(isAddingPayment || editingPayment) && (
        <Suspense
          fallback={
            <div className="text-center p-4">
              <Loader2 className="mx-auto h-6 w-6 animate-spin" />
            </div>
          }
        >
          <PaymentForm
            eventId={event.id}
            eventTitle={event.title}
            totalAmount={event.amount}
            paidAmount={event.paidAmount}
            eventCurrency={event.currency}
            onPaymentAdded={onPaymentAddedOrUpdated}
            onPaymentUpdated={onPaymentAddedOrUpdated}
            onCancel={() => {
              setIsAddingPayment(false);
              setEditingPayment(null);
            }}
            payment={editingPayment}
          />
        </Suspense>
      )}


      {/* Payments Timeline */}
      {payments.length === 0 ? (
        <Card className="border-dashed">
          <CardContent className="flex flex-col items-center justify-center py-10 text-center">
            <DollarSign className="h-12 w-12 text-muted-foreground/30 mb-4" />
            <p className="text-muted-foreground font-semibold">
              {t("booking.no_payment_history")}
            </p>
            <p className="text-sm text-muted-foreground mt-1">
              {t("booking.add_payment_to_start")}
            </p>
          </CardContent>
        </Card>
      ) : (
        <div className="relative border-l-2 border-primary/20 ml-6 pl-8">

                            paidAmount: payment.amount, // Override with specific payment amount
                            createdAt: payment.date, // Use payment date for receipt date
                          }}
                          type="receipt"
                        />

export function RescheduleEventDialog({
  open,
  onOpenChange,
  event,
  onConfirm,
}: RescheduleEventDialogProps) {
  const { t, lang } = useLang();
  const [date, setDate] = useState<Date | undefined>(undefined);
  const [startTime, setStartTime] = useState("");
  const [endTime, setEndTime] = useState("");
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState("");


  // Initialize state when modal opens
  useEffect(() => {
    if (open && event) {
      const start = new Date(event.startDate);
      const end = new Date(event.endDate);


      setDate(start);
      
      // Format times as HH:mm for the input type="time"
      setStartTime(format(start, "HH:mm"));
      setEndTime(format(end, "HH:mm"));
      setError("");
    }
  }, [open, event]);


  const handleConfirm = async () => {
    if (!date || !startTime || !endTime) {
      setError(t("booking.fill_all_fields") || "Please fill in all fields");
      return;
    }


    try {
      setIsLoading(true);
      setError("");


      // Construct new Date objects
      const [startHours, startMinutes] = startTime.split(":").map(Number);
      const [endHours, endMinutes] = endTime.split(":").map(Number);


      const newStartDate = new Date(date);
      newStartDate.setHours(startHours, startMinutes, 0, 0);


      const newEndDate = new Date(date);
      newEndDate.setHours(endHours, endMinutes, 0, 0);


      // Handle overnight events (if end time is before start time, assume it ends the next day)
      if (newEndDate <= newStartDate) {
        newEndDate.setDate(newEndDate.getDate() + 1);
      }


      await onConfirm(newStartDate, newEndDate);
      onOpenChange(false);
    } catch (err) {
      console.error("Failed to reschedule", err);
      setError(t("booking.update_failed") || "Failed to update event");
    } finally {
      setIsLoading(false);
    }
  };


  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="sm:max-w-[425px]">
        <DialogHeader>
          <DialogTitle>{t("booking.change_postponed")}</DialogTitle>
          <DialogDescription>
            {t("booking.reschedule_description")}
          </DialogDescription>
        </DialogHeader>

          {/* Date Picker */}
          <div className="flex flex-col gap-2">
            <Label>{t("booking.select_date")}</Label>
            <div className="border rounded-md p-2 flex justify-center">
              <Calendar
                mode="single"
                selected={date}
                onSelect={setDate}
                initialFocus
                className="rounded-md"
              />
            </div>
          </div>


          {/* Time Pickers */}
          <div className="grid grid-cols-2 gap-4">
            <div className="flex flex-col gap-2">
              <Label htmlFor="startTime">{t("booking.start_time")}</Label>

                <Clock className="absolute left-2.5 top-2.5 h-4 w-4 text-muted-foreground" />
                <Input

                />
              </div>
            </div>
            <div className="flex flex-col gap-2">
              <Label htmlFor="endTime">{t("booking.end_time")}</Label>

  setCustomStartDate,
  customEndDate,
  setCustomEndDate,
  formatDate,
  getCurrencySymbol,
  events,
  onStatCardClick,

  const [allJournalEntries, setAllJournalEntries] = useState<JournalEntry[]>(
    []
  );

  const [bankAccountId, setBankAccountId] = useState<string | null>(null);
  const [cashAccountId, setCashAccountId] = useState<string | null>(null);
  
  // Track loading states for specific data dependencies
  const [accountsLoading, setAccountsLoading] = useState(true);
  const [journalLoading, setJournalLoading] = useState(true);

      setAllJournalEntries(entries);
      setJournalLoading(false);
    });

  }, [companyId]);

    if (!companyId) return;
    const fetchAccounts = async () => {
      setAccountsLoading(true);
      try {
        const arAccountRes = await getAccountByName(

          "Accounts Receivable"

        if (arAccountRes.data) {
          setArAccountId(arAccountRes.data.id);

      case "today":
        start = new Date(now.setHours(0, 0, 0, 0));

      case "week":
        const first = now.getDate() - now.getDay();
        start = new Date(now.setDate(first));
        start.setHours(0, 0, 0, 0);
        end = new Date(now.setDate(first + 6));
        end.setHours(23, 59, 59, 999);
        break;
      case "month":

        break;
      case "custom":
        if (customStartDate && customEndDate) {
          // Force parsing as local time to match the Calendar picker
          start = new Date(customStartDate + 'T00:00:00');
          end = new Date(customEndDate + 'T23:59:59');

  const getEventsInDateRange = (useEndDate = false) => {
    const { start, end } = getDateRangeDates();


    if (!start || !end) {
      if (dateRange === "custom") return [];
      return events;
    }


    const finalStart = start;
    const finalEnd = end;


    return events.filter((event) => {
      const eventDate = useEndDate
        ? new Date(event.endDate)
        : new Date(event.startDate);
      return eventDate >= finalStart && eventDate <= finalEnd;

    // Critical: If custom is selected but dates aren't set, return zeroed data
    if (dateRange === "custom" && (!start || !end)) {

    }
  
    const totals: { [currency: string]: number } = {};
    const relevantEventIds = new Set<string>();
  
    allJournalEntries.forEach((entry) => {

      if (entry.date && typeof (entry.date as any).toDate === 'function') {
        entryDate = (entry.date as any).toDate();
      } else {
        entryDate = new Date(entry.date as any);
      }
  
      // 2. Perform strict range check
      if (start && end) {
        if (entryDate < start || entryDate > end) return;
      }
  
      if (entry.source !== "booking") return;
  
      entry.transactions.forEach((tx) => {
        if (tx.accountId === targetAccountId && tx.type === "debit") {
          const currency = entry.currency || "LYD";
          totals[currency] = (totals[currency] || 0) + tx.amount;
          if (entry.referenceId) {
            relevantEventIds.add(entry.referenceId);
          }
        }
      });
    });
  
    const totalByCurrency = Object.entries(totals).map(([currency, amount]) => ({
      currency,
      amount,
    }));
  
    const relevantEvents = events.filter((e) => relevantEventIds.has(e.id));
    return { totalByCurrency, relevantEvents };
  };


  const cashStats = useMemo(
    () => calculateRevenueStats(cashAccountId),

      dateRange,
      customStartDate,
      customEndDate,

  const bankStats = useMemo(

      dateRange,
      customStartDate,
      customEndDate,
      events,
    ]

      [eventId: string]: { balance: number; currency: string };
    } = {};


    allJournalEntries.forEach((entry) => {
      if (!entry.referenceId || entry.source !== "booking") return;


      if (!balances[entry.referenceId]) {
        const event = events.find((e) => e.id === entry.referenceId);
        balances[entry.referenceId] = {
          balance: 0,
          currency: event?.currency || "LYD",

          const amount = transaction.amount;
          if (entry.referenceId) {
            balances[entry.referenceId].balance +=
              transaction.type === "debit" ? amount : -amount;
          }
        }
      });


      if (entry.currency && entry.referenceId) {
        balances[entry.referenceId].currency = entry.currency;
      }
    });
    return balances;
  }, [arAccountId, allJournalEntries, events]);


  const calculatedExpectedPayments = useMemo(() => {
    const relevantEvents = getEventsInDateRange(true);
    const relevantEventIds = new Set(relevantEvents.map((e) => e.id));


    const totalExpectedByCurrency: { [currency: string]: number } = {};


    relevantEventIds.forEach((eventId) => {
      // Use balance from enriched events if available, otherwise fallback to calculated
      // The events prop is enriched now, so we can trust event.balance
      const event = events.find((e) => e.id === eventId);
      if (event && event.balance > 0.01) {
        const currency = event.currency || "LYD";
        if (!totalExpectedByCurrency[currency]) {
          totalExpectedByCurrency[currency] = 0;
        }
        totalExpectedByCurrency[currency] += event.balance;

function toLocalDateString(date: Date) {
  const year = date.getFullYear();
  const month = String(date.getMonth() + 1).padStart(2, "0");
  const day = String(date.getDate()).padStart(2, "0");
  return `${year}-${month}-${day}`;
}

// Rate limiting implementation
const createRateLimiter = (minInterval: number = 1000) => {
  let lastCall = 0;


  return () => {
    const now = Date.now();
    if (now - lastCall < minInterval) {
      throw new Error(
        `Rate limit exceeded. Please wait ${minInterval}ms between calls.`
      );
    }
    lastCall = now;
    return true;
  };
};


// Create rate limiters for different operations
const eventRateLimiter = createRateLimiter(1000);
const clientRateLimiter = createRateLimiter(500);
const placeRateLimiter = createRateLimiter(500);


// Default exchange rates
const DEFAULT_EXCHANGE_RATES: Record<string, number> = {
  USD: 1,
  LYD: 0.21,
  EUR: 1.08,
  GBP: 1.27,
  EGP: 0.032,
  SAR: 0.27,
  AED: 0.27,

// Convert Firestore Timestamp to Date
const convertTimestampsToDates = (data: any): any => {
  if (data && typeof data === "object") {
    if (data instanceof Timestamp) {
      return data.toDate();
    }
    if (Array.isArray(data)) {
      return data.map(convertTimestampsToDates);
    }
    const converted: any = {};
    for (const key in data) {
      converted[key] = convertTimestampsToDates(data[key]);
    }
    return converted;
  }
  return data;
};


// Convert Date to Firestore Timestamp for writing
const convertDatesToTimestamps = (data: any): any => {
  if (data && typeof data === "object") {
    if (data instanceof Date) {
      return Timestamp.fromDate(data);
    }
    if (Array.isArray(data)) {
      return data.map(convertDatesToTimestamps);
    }
    const converted: any = {};
    for (const key in data) {
      converted[key] = convertDatesToTimestamps(data[key]);
    }
    return converted;
  }
  return data;
};


// Calculate base currency amount from event currency
const calculateBaseCurrencyAmount = (
  amount: number,
  currency: string,
  baseCurrency: string = "USD"
): number => {
  if (currency === baseCurrency) return amount;


  const fromRate = DEFAULT_EXCHANGE_RATES[currency] || 1;
  const toRate = DEFAULT_EXCHANGE_RATES[baseCurrency] || 1;


  return amount * (fromRate / toRate);
};


// Calculate status based on paidAmount vs amount
const calculateStatus = (
  amount: number,
  balance: number
): "confirmed" | "pending" | "partially_paid" | "cancelled" => {
  const paidAmount = amount - balance;
  if (balance <= 0) return "confirmed"; // Fully paid or overpaid
  if (paidAmount > 0) return "partially_paid";
  if (paidAmount === 0) return "pending"; // No payment made yet
  return "pending";
};




// Convert currency amount
const convertCurrencyAmount = (
  amount: number,
  fromCurrency: string,
  toCurrency: string,
  customRate?: number
): number => {
  if (fromCurrency === toCurrency) return amount;


  const rate = customRate || getExchangeRate(fromCurrency, toCurrency);
  return amount * rate;
};


// Helper function to normalize currency codes
const normalizeCurrencyCode = (currency: string): string => {
  const currencyMap: Record<string, string> = {
    "د.ل": "LYD",
    "دينار ليبي": "LYD",
    دينار: "LYD",
    "ل.ل": "LYD",
    "د.ل.": "LYD",
    $: "USD",
    "€": "EUR",
    "£": "GBP",
    "¥": "JPY",
    "₹": "INR",
    "₽": "RUB",
    "₺": "TRY",
    R$: "BRL",
    C$: "CAD",
    A$: "AUD",
    NZ$: "NZD",
    "د.إ": "AED",
    "ر.س": "SAR",
    "ر.ق": "QAR",
    "р.ع": "OMR",
    "د.ك": "KWD",
    "د.ب": "BHD",
    "ج.م": "EGP",
    "د.م.": "MAD",
    "د.ت": "TND",
    "د.ج": "DZD",
    دولار: "USD",
    يورو: "EUR",
    "جنيه استرليني": "GBP",
    ين: "JPY",
    "دينار كويتي": "KWD",
    درهم: "AED",
    ريال: "SAR",
  };


  const cleanCurrency = currency.trim();


  // If it's already an ISO code (3 uppercase letters), return it
  if (/^[A-Z]{3}$/.test(cleanCurrency)) {
    return cleanCurrency;
  }


  // Try to map from our dictionary
  const mappedCurrency = currencyMap[cleanCurrency];
  if (mappedCurrency) {
    return mappedCurrency;
  }


  // If it looks like it could be an ISO code but not uppercase, uppercase it
  if (/^[a-zA-Z]{3}$/.test(cleanCurrency)) {
    return cleanCurrency.toUpperCase();
  }


  // Default to USD if we can't recognize it
  console.warn(`Unrecognized currency: "${currency}", defaulting to USD`);
  return "USD";
};


// Helper function to ensure we're storing proper ISO currency codes
const ensureCurrencyCode = (currency?: string): string => {
  if (!currency || currency.trim() === "") {
    return "USD";
  }
  return normalizeCurrencyCode(currency);
};


// Validate event data before saving
const validateEventData = (
  eventData: any
): { isValid: boolean; errors: string[] } => {
  const errors: string[] = [];


  if (!eventData.title || eventData.title.trim().length === 0) {
    errors.push("Title is required");
  }


  if (!eventData.customerName || eventData.customerName.trim().length === 0) {
    errors.push("Customer name is required");
  }


  if (eventData.amount === undefined || eventData.amount < 0) {
    errors.push("Valid amount is required");
  }


  if (!eventData.currency) {
    errors.push("Currency is required");
  }


  return {
    isValid: errors.length === 0,
    errors,
  };
};


export const createEventWithClientAndPlace = async (
  companyId: string,
  eventData: Omit<BookingEvent, "id" | "createdAt" | "updatedAt">,
  initialPaymentAmount: number = 0,
  clientData?: (Partial<Client> & { name: string }),
  placeData?: Omit<Place, "id">,
  paymentMethod?: string
): Promise<DataResponse<BookingEvent>> => {
  try {
    eventRateLimiter();


    const validation = validateEventData(eventData);
    if (!validation.isValid) {
      return { data: null, error: validation.errors.join(", ") };
    }


    const totalAmount = eventData.amount || 0;
    const paymentAmount = initialPaymentAmount || 0;


    if (paymentAmount > totalAmount) {
      return { data: null, error: "Initial payment cannot exceed total amount" };
    }


    // Determine the correct asset account for the payment
    const paymentAccountName =
      paymentMethod === "credit_card"
        ? "Credit Card Receipts"
        : paymentMethod === "bank_transfer"
        ? "Bank Transfers"
        : "Cash";


    const requiredAccountNames = [
      "Sales Revenue",
      "Accounts Receivable",
    ];
    if (paymentAmount > 0) {
        requiredAccountNames.push(paymentAccountName);
    }


    const accountData: { [name: string]: Account } = {};
    const accountTypes: { [name: string]: 'asset' | 'revenue' | 'expense' | 'liability' | 'equity' } = {
        "Sales Revenue": "revenue",
        "Accounts Receivable": "asset",
        "Cash": "asset",
        "Credit Card Receipts": "asset",
        "Bank Transfers": "asset",
    };


    for (const accName of requiredAccountNames) {
      let accountResult = await getAccountByName(companyId, accName);
      if (!accountResult.data) {
        console.log(`Creating missing default account: ${accName}`);
        const accType = accountTypes[accName];
        if (!accType) {
            throw new Error(`Account type for ${accName} is not defined.`);
        }
        const newAccountResult = await createAccount(companyId, { name: accName, type: accType, companyId: companyId });
        if (newAccountResult.data) {
          accountData[accName] = newAccountResult.data;
        } else {
          throw new Error(`Failed to create essential account: ${accName}`);
        }
      } else {
        accountData[accName] = accountResult.data;
      }
    }


    const normalizedCurrency = ensureCurrencyCode(eventData.currency);
    const baseCurrencyAmount = eventData.baseCurrencyAmount || calculateBaseCurrencyAmount(totalAmount, normalizedCurrency);
    const balance = totalAmount - paymentAmount;


    const safeEventData: Omit<BookingEvent, "id" | "createdAt" | "updatedAt"> = {
      ...(eventData as Omit<BookingEvent, "id" | "createdAt" | "updatedAt">),
      amount: totalAmount,
      balance: balance,
      currency: normalizedCurrency,
      baseCurrencyAmount: baseCurrencyAmount,
      status: calculateStatus(totalAmount, balance),
      source: eventData.source || "manual",
    };


    delete (safeEventData as any).paidAmount;
    const sanitizedEvent = stripUndefined(safeEventData);
    let eventId: string | null = null;


    await runTransaction(db, async (transaction) => {
      let clientId: string | null = null;
      let placeId: string | null = null;


      // Client handling logic
      if (clientData) {
        const clientsRef = collection(db, "companies", companyId, "clients");
        const clientPayload = { ...stripUndefined(clientData) };
        delete clientPayload.id;
        delete (clientPayload as Partial<Client>).balance; // Ensure balance is not set here


        let clientRef: DocumentReference;


        if (clientData.id) {
          clientRef = doc(clientsRef, clientData.id);
        } else {
          const clientQuery = query(clientsRef, where("name", "==", clientData.name.trim()), limit(1));
          const querySnapshot = await getDocs(clientQuery);
          if (!querySnapshot.empty) {
            clientRef = querySnapshot.docs[0].ref;
          } else {
            clientRef = doc(clientsRef);
          }
        }


        clientId = clientRef.id;
        const clientSnap = await transaction.get(clientRef);


        if (clientSnap.exists()) {
          // Client exists, update it (without touching balance)
          transaction.update(clientRef, { ...clientPayload, updatedAt: Timestamp.now() });
        } else {
          // Client does not exist, create it (without a balance field)
          transaction.set(clientRef, { ...clientPayload, createdAt: Timestamp.now(), updatedAt: Timestamp.now() });
        }
      }




      if (placeData) {
        const placesRef = collection(db, "companies", companyId, "places");
        const placeQuery = query(placesRef, where("name", "==", placeData.name.trim()), limit(1));
        const placeSnapshot = await getDocs(placeQuery);
        if (placeSnapshot.empty) {
          const newPlaceRef = doc(collection(db, "companies", companyId, "places"));
          placeId = newPlaceRef.id;
          transaction.set(newPlaceRef, { ...placeData, createdAt: Timestamp.now(), updatedAt: Timestamp.now() });
        } else {
          placeId = placeSnapshot.docs[0].id;
        }
      }


      const eventRef = doc(collection(db, "companies", companyId, "bookingEvents"));
      eventId = eventRef.id;
      transaction.set(eventRef, { ...convertDatesToTimestamps(sanitizedEvent), ...(clientId && { clientId }), ...(placeId && { placeId }), createdAt: Timestamp.now(), updatedAt: Timestamp.now() });


      // Journal Entry Creation
      const journalTransactions: Transaction[] = [];
      if (totalAmount > 0) {
        journalTransactions.push({
          accountId: accountData["Sales Revenue"].id,
          accountName: accountData["Sales Revenue"].name,
          amount: totalAmount,
          type: "credit",
        });


        if (paymentAmount > 0) {
          journalTransactions.push({
            accountId: accountData[paymentAccountName].id,
            accountName: accountData[paymentAccountName].name,
            amount: paymentAmount,
            type: "debit",
          });
        }


        if (balance > 0) {
          journalTransactions.push({
            accountId: accountData["Accounts Receivable"].id,
            accountName: accountData["Accounts Receivable"].name,
            amount: balance,
            type: "debit",
          });
        }
      }


      if (journalTransactions.length > 0) {
        const journalRef = doc(collection(db, "companies", companyId, "journalEntries"));
        transaction.set(journalRef, {
          date: Timestamp.now(),
          description: `دفعة أولى: ${sanitizedEvent.title}`,
          source: 'booking',
          referenceId: eventId,
          transactions: journalTransactions,
          companyId,
          currency: sanitizedEvent.currency,
          createdAt: Timestamp.now(),
          updatedAt: Timestamp.now(),
        });
      }
    });


    if (eventId) {
      const eventSnap = await getDoc(doc(db, "companies", companyId, "bookingEvents", eventId));
      if (eventSnap.exists()) {
        const createdEvent: BookingEvent = { id: eventId, ...convertTimestampsToDates(eventSnap.data()) } as BookingEvent;
        return { data: createdEvent, error: null };
      }
    }


    return { data: null, error: "Failed to create or retrieve event after transaction." };
  } catch (err) {
    console.error("Error creating booking event with transaction:", err);
    return { data: null, error: err instanceof Error ? err.message : String(err) };
  }
};

// Helper to apply transaction effect based on account type
const getBalanceChange = (
  transactionType: "debit" | "credit",
  accountType: "asset" | "expense" | "liability" | "equity" | "revenue",
  amount: number
): number => {
  if (accountType === "asset" || accountType === "expense") {
    // Debits increase, credits decrease
    return transactionType === "debit" ? amount : -amount;
  } else {
    // Liability, equity, revenue: Debits decrease, credits increase
    return transactionType === "credit" ? amount : -amount;
  }
};


export const createJournalEntry = async (
  companyId: string,
  entryData: JournalEntryData
): Promise<DataResponse<JournalEntry>> => {
  try {
    const debits = entryData.transactions
      .filter((t) => t.type === "debit")
      .reduce((sum, t) => sum + t.amount, 0);


    const credits = entryData.transactions
      .filter((t) => t.type === "credit")
      .reduce((sum, t) => sum + t.amount, 0);


    if (Math.abs(debits - credits) > 0.001) {
      return { data: null, error: "Debits do not equal credits." };
    }


    let newEntryId: string;
    const now = Timestamp.now();
    await runTransaction(db, async (transaction) => {
      // 1. Read Phase
      const accountDocs = new Map<string, DocumentSnapshot<DocumentData>>();
      for (const trans of entryData.transactions) {
        const accountRef = doc(
          db,
          "companies",
          companyId,
          "accounts",
          trans.accountId
        );
        const accountDoc = await transaction.get(accountRef);
        if (!accountDoc.exists()) {
          throw new Error(`Account with ID ${trans.accountId} not found.`);
        }
        accountDocs.set(trans.accountId, accountDoc);
      }


      // 2. Write Phase
      const journalRef = collection(
        db,
        "companies",
        companyId,
        "journalEntries"
      );
      const newEntryRef = doc(journalRef);
      newEntryId = newEntryRef.id;


      const entryToWrite = {
        ...entryData,
        companyId,
        date: entryData.date, // Use the Timestamp directly
        createdAt: now,
        updatedAt: now,
      };
      transaction.set(newEntryRef, entryToWrite);


      for (const trans of entryData.transactions) {
        const accountDoc = accountDocs.get(trans.accountId)!;
        const accountRef = accountDoc.ref;
        const currentBalance = accountDoc.data()!.balance || 0;
        const accountType = accountDoc.data()!.type;
        const balanceChange = getBalanceChange(
          trans.type,
          accountType,
          trans.amount
        );
        transaction.update(accountRef, {
          balance: currentBalance + balanceChange,
          updatedAt: now,
        });
      }
    });


    const newEntry: JournalEntry = {
      ...entryData,
      id: newEntryId!,
      companyId,
      createdAt: now,
      updatedAt: now,
    };
    return { data: newEntry, error: null };
  } catch (err: unknown) {
    console.error("Error creating journal entry:", err);
    return {
      data: null,
      error: err instanceof Error ? err.message : "An error occurred",
    };
  }
};

      .filter((t) => t.type === "debit")

    const credits = entryData.transactions
      .filter((t) => t.type === "credit")
      .reduce((sum, t) => sum + t.amount, 0);


    if (Math.abs(debits - credits) > 0.001) {

      );


      // 1. Read Phase
      const oldJournalEntryDoc = await transaction.get(journalEntryRef);
      if (!oldJournalEntryDoc.exists()) {
        throw new Error("Journal entry not found.");
      }
      const oldTransactions = oldJournalEntryDoc.data()
        .transactions as Transaction[];


      const allAccountIds = new Set([
        ...oldTransactions.map((t) => t.accountId),
        ...entryData.transactions.map((t) => t.accountId),
      ]);

          ref: DocumentReference<DocumentData>;

        );
        const accountDoc = await transaction.get(accountRef);
        if (!accountDoc.exists())
          throw new Error(`Account ${accountId} not found.`);
        accountData.set(accountId, { ref: accountRef, doc: accountDoc });
      }


      // 2. Calculation & Write Phase
      const balanceChanges = new Map<string, number>();
      const now = Timestamp.now();


      // Revert old transactions
      for (const trans of oldTransactions) {
        const accountType = accountData.get(trans.accountId)!.doc.data()!.type;
        const change = getBalanceChange(trans.type, accountType, trans.amount);
        balanceChanges.set(
          trans.accountId,
          (balanceChanges.get(trans.accountId) || 0) - change
        );
      }


      // Apply new transactions
      for (const trans of entryData.transactions) {
        const accountType = accountData.get(trans.accountId)!.doc.data()!.type;
        const change = getBalanceChange(trans.type, accountType, trans.amount);
        balanceChanges.set(
          trans.accountId,
          (balanceChanges.get(trans.accountId) || 0) + change
        );
      }


      // Update account balances
      for (const [accountId, change] of balanceChanges.entries()) {
        const account = accountData.get(accountId)!;
        const currentBalance = account.doc.data()!.balance || 0;
        transaction.update(account.ref, {
          balance: currentBalance + change,
          updatedAt: now,
        });
      }


      // Update journal entry
      transaction.update(journalEntryRef, {
        ...entryData,
        date: entryData.date, // Use the Timestamp directly
        updatedAt: now,
      });
    });


    return { data: null, error: null };
  } catch (err: unknown) {
    console.error("Error updating journal entry:", err);
    return {
      data: null,
      error: err instanceof Error ? err.message : "An error occurred",
    };
  }
};


export const deleteJournalEntry = async (
  companyId: string,
  journalEntryId: string
): Promise<DataResponse<null>> => {
  try {
    await runTransaction(db, async (transaction) => {
      const journalEntryRef = doc(
        db,
        "companies",
        companyId,
        "journalEntries",
        journalEntryId
      );


      // 1. Read Phase
      const journalEntryDoc = await transaction.get(journalEntryRef);
      if (!journalEntryDoc.exists()) {
        // If it's already deleted, the job is done.
        console.log("Journal entry already deleted.");
        return;
      }
      const journalEntry = journalEntryDoc.data() as JournalEntry;


      // 2. Revert account balances
      for (const trans of journalEntry.transactions) {
        const accountRef = doc(
          db,
          "companies",
          companyId,
          "accounts",
          trans.accountId
        );
        const accountDoc = await transaction.get(accountRef);


        if (!accountDoc.exists()) {
          throw new Error(`Account with ID ${trans.accountId} not found.`);
        }


        const currentBalance = accountDoc.data().balance || 0;
        const accountType = accountDoc.data().type;


        // Revert the change
        const balanceChange = getBalanceChange(
          trans.type,
          accountType,
          trans.amount
        );
        transaction.update(accountRef, {
          balance: currentBalance - balanceChange,
          updatedAt: Timestamp.now(),
        });
      }


      // 3. Delete the journal entry
      transaction.delete(journalEntryRef);
    });


    return { data: null, error: null };
  } catch (err: unknown) {
    console.error("Error deleting journal entry:", err);
    return {
      data: null,
      error: err instanceof Error ? err.message : "An error occurred",
    };
  }
};
